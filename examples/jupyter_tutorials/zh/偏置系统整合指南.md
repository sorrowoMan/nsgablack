- **不确定性偏置**：基于不确定性探索

#### 2. 混合偏置系统
```python
class BayesianHybridBias:
    """贝叶斯混合偏置系统"""
    def __init__(self):
        self.gp_bias = GaussianProcessBias()
        self.domain_bias = DomainKnowledgeBias()
        self.constraint_bias = ConstraintBias()

    def apply(self, x, context):
        # 组合多种偏置
        gp_guidance = self.gp_bias.predict(x)
        domain_knowledge = self.domain_bias.apply(x)
        constraint_adjustment = self.constraint_bias.apply(x)

        # 加权融合
        combined_bias = self._combine_biases(
            gp_guidance, domain_knowledge, constraint_adjustment
        )
        return combined_bias
```

class AdaptiveNeighborhoodBias(BiasBase):
    """自适应邻域偏置"""
    def apply_bias(self, solution, context):
        # 基于搜索阶段调整邻域
        if context['iteration'] < 0.3 * context['max_iterations']:
            # 早期：大邻域探索
            return self._large_neighborhood(solution)
        else:
            # 后期：小邻域精细搜索
